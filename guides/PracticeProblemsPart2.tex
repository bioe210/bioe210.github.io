\begin{problem}{8.2}

\begin{center}
	\begin{tabular}{cc}
		\toprule
		$x$ & $y$ \\
		\midrule
		1 & 4 \\
		2 & 5 \\
		-1 & 2 \\
		\bottomrule	
	\end{tabular}
