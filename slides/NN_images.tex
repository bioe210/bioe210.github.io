\newcommand\layerskip{1}
\newcommand\neuronskip{1}
\begin{tikzpicture}
	\tikzstyle{neuron-input}=[circle,fill=blue,minimum size=0.5cm]
	\tikzstyle{neuron-hidden}=[circle,fill=black,minimum size=0.5cm]
	\tikzstyle{neuron-output}=[circle,fill=orange,minimum size=0.5cm]
	
	\def\mynodes{3/input,4/hidden,4/hidden,4/hidden,4/hidden,5/output}
	
	
	\foreach \n / \cls [count=\lyr] in \mynodes {
		\pgfmathsetmacro\ystart{(\n-1)*\neuronskip/2}
		\foreach \nd in {1,...,\n} {
			\pgfmathsetmacro\x{(\lyr-1)*\layerskip}
			\pgfmathsetmacro\y{\ystart - (\nd-1)*\neuronskip}
			\node[neuron-\cls] (N-\lyr-\nd) at (\x,\y) {};
		}
	}
	
	\foreach \n / \cls [count=\lyr, remember=\n as \prevn] in \mynodes {
		\ifthenelse{\lyr > 1}{
	  		\foreach \np in {1,...,\prevn} {
	  			\foreach \nc in {1,...,\n} {
	  					\pgfmathtruncatemacro\prevlyr{\lyr-1}
	  					\draw (N-\prevlyr-\np) -- (N-\lyr-\nc);
	  			}
	  		}
  		}{}
	}
	
\end{tikzpicture}
